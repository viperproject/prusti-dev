use super::{expression::Expression, ty::Type, variable::VariableDecl};
use crate::common::display;
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "function {}<{}>({}): {}\n{}{}{{ {} }}\n",
    name,
    "display::cjoin(type_arguments)",
    "display::cjoin(parameters)",
    return_type,
    "display::foreach!(\"  requires {}\n\", pres)",
    "display::foreach!(\"  ensures {}\n\", pres)",
    "display::option!(body, \"{{ {} }}\n\", \"\")"
)]
pub struct FunctionDecl {
    pub name: String,
    pub type_arguments: Vec<Type>,
    pub parameters: Vec<VariableDecl>,
    pub return_type: Type,
    pub pres: Vec<Expression>,
    pub posts: Vec<Expression>,
    pub body: Option<Expression>,
}
impl PartialEq for FunctionDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
            && self.type_arguments == other.type_arguments
            && self.parameters == other.parameters
            && self.return_type == other.return_type
            && self.pres == other.pres
            && self.posts == other.posts
            && self.body == other.body
    }
}
impl core::hash::Hash for FunctionDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.type_arguments.hash(state);
        self.parameters.hash(state);
        self.return_type.hash(state);
        self.pres.hash(state);
        self.posts.hash(state);
        self.body.hash(state);
    }
}
impl Ord for FunctionDecl {
    #[allow(unused_variables)]
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        self.name
            .cmp(&other.name)
            .then(self.type_arguments.cmp(&other.type_arguments))
            .then(self.parameters.cmp(&other.parameters))
            .then(self.return_type.cmp(&other.return_type))
            .then(self.pres.cmp(&other.pres))
            .then(self.posts.cmp(&other.posts))
            .then(self.body.cmp(&other.body))
    }
}
impl PartialOrd for FunctionDecl {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}
impl FunctionDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<
        G0: Into<String>,
        G1: Into<Vec<Type>>,
        G2: Into<Vec<VariableDecl>>,
        G3: Into<Type>,
        G4: Into<Vec<Expression>>,
        G5: Into<Vec<Expression>>,
        G6: Into<Option<Expression>>,
    >(
        name: G0,
        type_arguments: G1,
        parameters: G2,
        return_type: G3,
        pres: G4,
        posts: G5,
        body: G6,
    ) -> Self {
        Self {
            name: name.into(),
            type_arguments: type_arguments.into(),
            parameters: parameters.into(),
            return_type: return_type.into(),
            pres: pres.into(),
            posts: posts.into(),
            body: body.into(),
        }
    }
}
impl FunctionDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<
        G0: Into<String>,
        G1: Into<Vec<Type>>,
        G2: Into<Vec<VariableDecl>>,
        G3: Into<Type>,
        G4: Into<Vec<Expression>>,
        G5: Into<Vec<Expression>>,
        G6: Into<Option<Expression>>,
    >(
        name: G0,
        type_arguments: G1,
        parameters: G2,
        return_type: G3,
        pres: G4,
        posts: G5,
        body: G6,
    ) -> Self {
        Self {
            name: name.into(),
            type_arguments: type_arguments.into(),
            parameters: parameters.into(),
            return_type: return_type.into(),
            pres: pres.into(),
            posts: posts.into(),
            body: body.into(),
        }
    }
}

use super::super::ast::{
    expression::{Expression, Local},
    statement::Statement,
};
use crate::common::{check_mode::CheckMode, display};
use std::collections::BTreeMap;
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "procedure({}) {}\n{{\n{}}}",
    check_mode,
    name,
    "display::foreach2!(\"    label {}\n{}\", basic_blocks.keys(), basic_blocks.values())"
)]
pub struct ProcedureDecl {
    pub name: String,
    pub check_mode: CheckMode,
    pub entry: BasicBlockId,
    pub exit: BasicBlockId,
    pub basic_blocks: BTreeMap<BasicBlockId, BasicBlock>,
}
#[derive(
    Debug,
    derive_more :: Display,
    Clone,
    serde :: Serialize,
    serde :: Deserialize,
    Eq,
    PartialOrd,
    Ord,
    derive_more :: Constructor,
    derive_more :: AsRef,
)]
pub struct BasicBlockId {
    pub name: String,
}
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "{}\n    {}\n",
    "display::foreach!(\"    {}\n\", statements)",
    successor
)]
pub struct BasicBlock {
    pub statements: Vec<Statement>,
    pub successor: Successor,
}
#[derive(
    Debug,
    derive_more :: Display,
    Clone,
    serde :: Serialize,
    serde :: Deserialize,
    PartialEq,
    Eq,
    Hash,
)]
pub enum Successor {
    Exit,
    Goto(BasicBlockId),
    #[display(fmt = "switch")]
    GotoSwitch(Vec<(Expression, BasicBlockId)>),
    #[display(fmt = "non-det-switch")]
    NonDetChoice(BasicBlockId, BasicBlockId),
}
impl PartialEq for ProcedureDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
            && self.check_mode == other.check_mode
            && self.entry == other.entry
            && self.exit == other.exit
            && self.basic_blocks == other.basic_blocks
    }
}
impl core::hash::Hash for ProcedureDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.check_mode.hash(state);
        self.entry.hash(state);
        self.exit.hash(state);
        self.basic_blocks.hash(state);
    }
}
impl PartialEq for BasicBlockId {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
    }
}
impl core::hash::Hash for BasicBlockId {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
    }
}
impl PartialEq for BasicBlock {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.statements == other.statements && self.successor == other.successor
    }
}
impl core::hash::Hash for BasicBlock {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.statements.hash(state);
        self.successor.hash(state);
    }
}

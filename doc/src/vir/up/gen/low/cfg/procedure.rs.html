use crate::{
    common::display,
    low::ast::{expression::Expression, statement::Statement, variable::VariableDecl},
};
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "procedure {} {{\n{}\n{}}}\n",
    name,
    "display::foreach!(\"  var {};\n\", locals)",
    "display::foreach!(\"{}\n\", basic_blocks)"
)]
pub struct ProcedureDecl {
    pub name: String,
    pub locals: Vec<VariableDecl>,
    pub basic_blocks: Vec<BasicBlock>,
}
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(
    fmt = "  block {} {{\n{}\n    {}\n  }}\n",
    label,
    "display::foreach!(\"    {};\n\", statements)",
    successor
)]
pub struct BasicBlock {
    pub label: Label,
    pub statements: Vec<Statement>,
    pub successor: Successor,
}
#[derive(Debug, derive_more :: Display, Clone, serde :: Serialize, serde :: Deserialize, Eq)]
#[display(fmt = "label {}", name)]
pub struct Label {
    pub name: String,
}
#[derive(
    Debug,
    derive_more :: Display,
    Clone,
    serde :: Serialize,
    serde :: Deserialize,
    PartialEq,
    Eq,
    Hash,
)]
pub enum Successor {
    Return,
    Goto(Label),
    #[display(fmt = "switch")]
    GotoSwitch(Vec<(Expression, Label)>),
}
impl PartialEq for ProcedureDecl {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
            && self.locals == other.locals
            && self.basic_blocks == other.basic_blocks
    }
}
impl core::hash::Hash for ProcedureDecl {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
        self.locals.hash(state);
        self.basic_blocks.hash(state);
    }
}
impl ProcedureDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<String>, G1: Into<Vec<VariableDecl>>, G2: Into<Vec<BasicBlock>>>(
        name: G0,
        locals: G1,
        basic_blocks: G2,
    ) -> Self {
        Self {
            name: name.into(),
            locals: locals.into(),
            basic_blocks: basic_blocks.into(),
        }
    }
}
impl ProcedureDecl {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<
        G0: Into<String>,
        G1: Into<Vec<VariableDecl>>,
        G2: Into<Vec<BasicBlock>>,
    >(
        name: G0,
        locals: G1,
        basic_blocks: G2,
    ) -> Self {
        Self {
            name: name.into(),
            locals: locals.into(),
            basic_blocks: basic_blocks.into(),
        }
    }
}
impl PartialEq for BasicBlock {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.label == other.label
            && self.statements == other.statements
            && self.successor == other.successor
    }
}
impl core::hash::Hash for BasicBlock {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.label.hash(state);
        self.statements.hash(state);
        self.successor.hash(state);
    }
}
impl BasicBlock {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<Label>, G1: Into<Vec<Statement>>, G2: Into<Successor>>(
        label: G0,
        statements: G1,
        successor: G2,
    ) -> Self {
        Self {
            label: label.into(),
            statements: statements.into(),
            successor: successor.into(),
        }
    }
}
impl BasicBlock {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<G0: Into<Label>, G1: Into<Vec<Statement>>, G2: Into<Successor>>(
        label: G0,
        statements: G1,
        successor: G2,
    ) -> Self {
        Self {
            label: label.into(),
            statements: statements.into(),
            successor: successor.into(),
        }
    }
}
impl PartialEq for Label {
    #[allow(unused_variables)]
    fn eq(&self, other: &Self) -> bool {
        self.name == other.name
    }
}
impl core::hash::Hash for Label {
    #[allow(unused_variables)]
    fn hash<H: core::hash::Hasher>(&self, state: &mut H) {
        self.name.hash(state);
    }
}
impl Label {
    #[allow(clippy::too_many_arguments)]
    pub fn new<G0: Into<String>>(name: G0) -> Self {
        Self { name: name.into() }
    }
}
impl Label {
    #[allow(clippy::too_many_arguments)]
    pub fn new_with_pos<G0: Into<String>>(name: G0) -> Self {
        Self { name: name.into() }
    }
}

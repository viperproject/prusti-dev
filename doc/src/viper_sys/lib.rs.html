#![deny(unused_must_use)]

#[rustfmt::skip]
#[path = "../gen/mod.rs"]
pub mod wrappers;

use jni::{errors::Result, objects::JObject, JNIEnv};

pub fn get_system_out<'a>(env: &'a JNIEnv) -> Result<JObject<'a>> {
    env.get_static_field("java/lang/System", "out", "Ljava/io/PrintStream;")
        .and_then(|x| x.l())
}

A marker trait to ensure proper types are used for custom rejections.

Example

use warp::{Filter, reject::Reject};

#[derive(Debug)]
struct RateLimited;

impl Reject for RateLimited {}

let route = warp::any().and_then(|| async {
    Err::<(), _>(warp::reject::custom(RateLimited))
});

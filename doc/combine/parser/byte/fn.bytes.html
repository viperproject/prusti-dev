Parses the bytes s.

If you have a stream implementing RangeStream such as &[u8] you can also use the range parser which may be more efficient.

let result = bytes(&b"rust"[..])
    .parse(&b"rust"[..])
    .map(|x| x.0);
assert_eq!(result, Ok(&b"rust"[..]));
